          `ID` 
          ,`Conto`
          ,`DataMovimento` AS DataMovimento 
          ,`TipoMovimento` AS Pagamento 
          ,`Causale`

          ,`Descrizione` 
          ,`Transazione` 
          ,`Consolidato` 
        FROM viewmovimenti 
         WHERE 1=1 AND (ID_TRANSAZIONE IS NULL)  AND (`viewmovimenti`.`Segno` = '-')   AND (`viewmovimenti`.`ID_CAUSALE_MOVIMENTO` IN (1))  AND YEAR(`viewmovimenti`.DataMovimento) = 2015  AND MONTH(`viewmovimenti`.DataMovimento) = 9 
        ORDER BY DataMovimento DESC, ID DESC
